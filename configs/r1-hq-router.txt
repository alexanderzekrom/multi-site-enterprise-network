! ============================================================================
! Multi-Site Enterprise Network - HQ Router (R1) Configuration
! ============================================================================
! Version: 1.0
! Platform: Cisco IOS Router
! Features: OSPF Area 0, DHCP, NAT, Inter-VLAN Routing
! ============================================================================

! Basic Configuration
hostname R1-HQ
enable secret 12345
service password-encryption
no ip domain-lookup

! Security Configuration
username admin privilege 15 secret 1234

line console 0
 password 1234
 login

line vty 0 4
 password 1234
 login
 transport input ssh
 access-class 10 in

! SSH Configuration
ip domain-name enterprise.local
crypto key generate rsa modulus 1024

! Management Access
access-list 10 permit 10.0.10.0 0.0.0.255

! ============================================================================
! INTERFACE CONFIGURATION
! ============================================================================

! Trunk Interface for VLANs (Router-on-a-Stick)
interface GigabitEthernet0/0
 description Trunk to HQ Switch
 no shutdown
 ip nat inside

! IT VLAN Interface
interface GigabitEthernet0/0.10
 description IT Department VLAN
 encapsulation dot1Q 10
 ip address 10.0.10.1 255.255.255.0
 ip nat inside

! Sales VLAN Interface
interface GigabitEthernet0/0.20
 description Sales Department VLAN
 encapsulation dot1Q 20
 ip address 10.0.20.1 255.255.255.0
 ip nat inside
 ip access-group 101 in

! Admin VLAN Interface
interface GigabitEthernet0/0.30
 description Admin Department VLAN
 encapsulation dot1Q 30
 ip address 10.0.30.1 255.255.255.0
 ip nat inside
 ip access-group 101 in

! Internet/WAN Interface
interface GigabitEthernet0/1
 description Internet Connection
 ip address 200.1.1.2 255.255.255.252
 ip nat outside
 no shutdown

! Branch 1 Link
interface GigabitEthernet0/2
 description Link to Branch 1 (R3)
 ip address 10.0.1.1 255.255.255.252
 no shutdown

! Branch 2 Link
interface GigabitEthernet0/3
 description Link to Branch 2 (R4)
 ip address 10.0.2.1 255.255.255.252
 no shutdown

! ============================================================================
! DHCP CONFIGURATION
! ============================================================================

! DHCP Exclusions
ip dhcp excluded-address 10.0.10.1 10.0.10.10
ip dhcp excluded-address 10.0.20.1 10.0.20.10
ip dhcp excluded-address 10.0.30.1 10.0.30.10

! IT Department DHCP Pool
ip dhcp pool IT
 network 10.0.10.0 255.255.255.0
 default-router 10.0.10.1
 dns-server 8.8.8.8
 lease 7

! Sales Department DHCP Pool
ip dhcp pool SALES
 network 10.0.20.0 255.255.255.0
 default-router 10.0.20.1
 dns-server 8.8.8.8
 lease 7

! Admin Department DHCP Pool
ip dhcp pool ADMIN
 network 10.0.30.0 255.255.255.0
 default-router 10.0.30.1
 dns-server 8.8.8.8
 lease 7

! ============================================================================
! OSPF ROUTING CONFIGURATION
! ============================================================================

router ospf 1
 router-id 1.1.1.1
 network 10.0.10.0 0.0.0.255 area 0
 network 10.0.20.0 0.0.0.255 area 0
 network 10.0.30.0 0.0.0.255 area 0
 network 10.0.1.0 0.0.0.3 area 0
 network 10.0.2.0 0.0.0.3 area 0

! ============================================================================
! NAT CONFIGURATION
! ============================================================================

! NAT ACL for Internal Networks
access-list 1 permit 10.0.0.0 0.255.255.255

! NAT Overload Configuration
ip nat inside source list 1 interface GigabitEthernet0/1 overload

! ============================================================================
! ACCESS CONTROL LISTS
! ============================================================================

! Inter-VLAN Security - Block Sales-Admin Communication
access-list 101 deny ip 10.0.20.0 0.0.0.255 10.0.30.0 0.0.0.255
access-list 101 deny ip 10.0.30.0 0.0.0.255 10.0.20.0 0.0.0.255
access-list 101 permit ip any any

! ============================================================================
! ROUTING AND SERVICES
! ============================================================================

! Default Route to Internet
ip route 0.0.0.0 0.0.0.0 200.1.1.1

! Enable HTTP/HTTPS Management (Optional)
ip http server
ip http secure-server

! ============================================================================
! BANNER AND MESSAGES
! ============================================================================

banner motd ^C
*****************************************************************
*        MULTI-SITE ENTERPRISE NETWORK - HQ ROUTER             *
*        Unauthorized Access Prohibited                        *
*****************************************************************
^C

! ============================================================================
! END OF CONFIGURATION
! ============================================================================

! Save Configuration
write memory
